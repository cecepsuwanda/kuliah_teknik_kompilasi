\section{Checklist Kualitas Kode}

\subsection{Checklist untuk Lexer}

\begin{itemize}
  \item[$\square$] Regular expressions terdefinisi dengan benar
  \item[$\square$] Token recognition akurat
  \item[$\square$] Error handling untuk invalid input
  \item[$\square$] Memory management yang benar
  \item[$\square$] Kode style konsisten
  \item[$\square$] Dokumentasi lengkap
  \item[$\square$] Testing coverage memadai
\end{itemize}

\subsection{Checklist untuk Parser}

\begin{itemize}
  \item[$\square$] Grammar definition benar
  \item[$\square$] Parsing algorithm efisien
  \item[$\square$] Syntax error reporting jelas
  \item[$\square$] AST construction benar
  \item[$\square$] Error recovery mechanism
  \item[$\square$] Testing dengan valid dan invalid input
\end{itemize}

\subsection{Checklist untuk Code Generator}

\begin{itemize}
  \item[$\square$] Instruction selection optimal
  \item[$\square$] Register allocation efisien
  \item[$\square$] Target code generation benar
  \item[$\square$] Optimization terimplementasi
  \item[$\square$] Performance testing dilakukan
  \item[$\square$] Cross-platform compatibility
\end{itemize}
