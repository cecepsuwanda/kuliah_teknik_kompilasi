\subsection{Case Studies}

\begin{lstlisting}[language=C]
// Case study: Parser optimization
// Before optimization
int parse_slow(char *input) {
    // Multiple function calls
    token_t token1 = get_next_token(input);
    token_t token2 = get_next_token(token1->remaining);
    token_t token3 = get_next_token(token2->remaining);
    return process_tokens(token1, token2, token3);
}

// After optimization
int parse_fast(char *input) {
    // Inlined token processing
    token_t tokens[3];
    int count = 0;
    
    while (*input && count < 3) {
        tokens[count++] = get_next_token_inline(&input);
    }
    
    return process_tokens_inline(tokens[0], tokens[1], tokens[2]);
}
\end{lstlisting}
