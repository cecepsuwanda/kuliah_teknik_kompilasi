\section{Structure and Union Layout}

\subsection{Structure Memory Layout}

\begin{lstlisting}[language=C]
struct Student {
    char name[50];      // 50 bytes
    int age;            // 4 bytes (aligned to 4)
    double gpa;         // 8 bytes (aligned to 8)
    char grade;         // 1 byte
    // Padding: 3 bytes for alignment
};

// Total size: 50 + 4 + 4(padding) + 8 + 1 + 3(padding) = 70 bytes
\end{lstlisting}

\subsection{Union Memory Layout}

\begin{lstlisting}[language=C]
union Data {
    int i;              // 4 bytes
    float f;            // 4 bytes  
    char c[4];          // 4 bytes
    double d;           // 8 bytes (largest member)
};

// Union size = size of largest member = 8 bytes
\end{lstlisting}
