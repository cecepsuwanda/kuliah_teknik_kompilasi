\subsection{Benchmarking}

\begin{lstlisting}[language=C]
// Simple benchmark framework
#include <time.h>
#include <stdio.h>

void benchmark_compiler(char *input_file) {
    clock_t start = clock();
    
    // Run compiler
    compile_file(input_file);
    
    clock_t end = clock();
    double elapsed = ((double)(end - start)) / CLOCKS_PER_SEC;
    
    printf("Compilation time: %.3f seconds\n", elapsed);
}
\end{lstlisting}
